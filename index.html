// --- SIDEKICK LOGIC --- async function addSidekick() { const target = document.getElementById('sidekick-search').value; if(!target) return; // Lookup user by username const { data: friend, error: fErr } = await _supabase.from('profiles').select('user_id').eq('username', target).single(); if(fErr || !friend) return alert("System not found!"); const { data: { user } } = await _supabase.auth.getUser(); // Add connection const { error: linkErr } = await _supabase.from('connections').insert([{ user_id: user.id, friend_id: friend.user_id }]); if(linkErr) alert("Already buddies or error!"); else { alert("Added!"); loadSidekicks(); } } async function loadSidekicks() { const { data: { user } } = await _supabase.auth.getUser(); // JOIN query to get profile data along with the connection const { data, error } = await _supabase.from('connections').select(` friend_id, status, profiles:friend_id ( username, system_data ) `).eq('user_id', user.id); if(error) return console.error(error); const list = document.getElementById('sidekicks-list'); list.innerHTML = data.length ? '' : 'No sidekicks added yet.'; data.forEach(item => { const buddy = item.profiles; const buddyData = buddy.system_data || {}; const friendMembers = buddyData.members || []; const frontingList = friendMembers.filter(m => m.isFronting); const card = document.createElement('div'); card.className = 'sidekick-card'; card.style.cursor = 'pointer'; card.style.transition = '0.2s'; // Clicking the card opens their full profile in a new tab card.onclick = () => { const profileWindow = window.open('', '_blank'); profileWindow.document.write(` ${buddy.username}'s System

${buddyData.name || buddy.username}

${buddyData.bio || 'No bio provided.'}

${friendMembers.map(m => `

${m.name} ${m.isFronting ? '⭕' : ''}

Pronouns: ${m.pronouns}

Role: ${m.role}

Bio: ${m.bio}

`).join('')}
`); }; // Add hover effect via JS card.onmouseover = () => card.style.borderColor = '#ff00ff'; card.onmouseout = () => card.style.borderColor = '#ffdeeb'; card.innerHTML = `
${buddy.username}
${buddyData.name || 'Personal System'}
Currently: ${frontingList.length ? frontingList.map(m => m.name).join(', ') : 'No one fronting'}
Click to view full profile →
`; list.appendChild(card); }); }
